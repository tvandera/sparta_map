876 TickQuantum* determineTickQuantum_(Tick rel_time)
877 {
878 Tick index_time = (current_tick_ + rel_time);
879
880 // This might look inefficient, but 99.9% of the time the
881 // event being scheduled is either on the current time
882 // quantum or the next. A straight walk of two elements
883 // is faster than a map lookup.
884 TickQuantum * rit = current_tick_quantum_;
885 TickQuantum * last_tq = nullptr;
886 while(rit != nullptr) {
887 if(rit->tick == index_time) {
888 // This is the time quantum to add the event to
889 break;
890 }
891 else if(rit->tick > index_time) {
892 // We're past the tick quantum. Insert before rit
893 rit = tick_quantum_allocator_.create(firing_group_count_);
894 rit->tick = index_time;
895
896 // rit could have pointed to the
897 // current_tick_quantum_. If so, move the
898 // current_tick_quantum_ back.
899 if(SPARTA_EXPECT_FALSE(last_tq == nullptr)) {
900 rit->next = current_tick_quantum_;
901 current_tick_quantum_ = rit;
902 }
903 else {
904 rit->next = last_tq->next;
905 last_tq->next = rit;
906 }
907 break;
908 }
909 last_tq = rit;
910 rit = rit->next;
911 }
912 // If we reached the end of our search in the time quantums,
913 // we need to append this event to the end of the list
914 //if(SPARTA_EXPECT_FALSE(rit == nullptr))
915 if(rit == nullptr)
916 {
917 rit = tick_quantum_allocator_.create(firing_group_count_);
918 rit->tick = index_time;
919 if(SPARTA_EXPECT_TRUE(last_tq != nullptr)) {
920 last_tq->next = rit;
921 }
922 else {
923 // If the last_tq was null, there is a good change the
924 // current_tick_quantum_ is null as well
925 if(current_tick_quantum_ == nullptr) {
926 current_tick_quantum_ = rit;
927 }
928 }
929 }
930
931 return rit;
932 }

1130inline void Scheduler::printNextCycleEventTree(StreamType& os,
1131 uint32_t curr_grp,
1132 uint32_t curr_event,
1133 uint32_t future) const
1134{
1135 if(current_tick_quantum_ == nullptr) {
1136 os << "sparta::Scheduler is empty" << std::endl;
1137 return;
1138 }
1139
1140 uint32_t scheduler_map_idx = current_tick_ + future;
1141 os << "Scheduler's event tree for tick: " << scheduler_map_idx << std::endl;
1142 if(current_tick_quantum_->tick > scheduler_map_idx) {
1143 os << "\tNo events for time: '"
1144 << scheduler_map_idx << "' next event @"
1145 << current_tick_quantum_->tick << std::endl;
1146 }
1147 const TickQuantum::Groups & group_array = current_tick_quantum_->groups;
1148 for(uint32_t i = curr_grp; i < group_array.size(); ++i)
1149 {
1150 std::stringstream output;
1151 if((i + 1) == group_array.size()) {
1152 output << "\tGroup[zero]: ";
1153 }
1154 else {
1155 output << "\tGroup[" << i + 1 << "]: ";
1156 }
1157 const TickQuantum::Scheduleables & scheduleables = group_array[i];
1158
1159 output << SPARTA_CURRENT_COLOR_GREEN;
1160 for(uint32_t x = 0; x < scheduleables.size(); ++x)
1161 {
1162 if(x) {
1163 output << ", ";
1164 }
1165 if((curr_grp == i) && (curr_event == x)) {
1166 output << SPARTA_CURRENT_COLOR_BRIGHT_GREEN << getScheduleableLabel_(scheduleables[x]);
1167 }
1168 else {
1169 output << SPARTA_CURRENT_COLOR_GREEN << getScheduleableLabel_(scheduleables[x]);
1170 }
1171 }
1172
1173 os << output.str() << SPARTA_CURRENT_COLOR_NORMAL << std::endl;
1174 }
1175}
