40 inline Clock::Cycle calculateSpartaOffset(const sparta::Clock * sparta_clk,
41 sc_core::sc_time::value_type sysc_offset)
42 {
43 //
44 // This is a transaction coming from SysC that is on SysC's
45 // clock, not Sparta's. Need to find the same tick cycle on
46 // the Sparta clock and align the time for the transaction.
47 // Keep in mind that Sparta's scheduler starts on tick 1, not
48 // 0 like SysC.
49 //
50 // For example,
51 // - The Sparta's clock is at 7 ticks (6 from SysC POV, hence the - 1)
52 // - The SysC clock is at 10 ticks
53 // - The transaction's delay is 1 tick (to be fired at tick 11)
54 //
55 // sysc_clock - sparta_clock + delay = 4 cycles on sparta clock (11)
56 //
57
58 // Send to memory with the given delay - NS -> clock cycles.
59 // The Clock is on the same freq as the memory block
60 auto current_sc_time = sc_core::sc_time_stamp().value();
61 const auto current_tick = sparta_clk->currentTick() - 1;
62 sparta_assert(sc_core::sc_time_stamp().value() >= current_tick);
63 const auto final_relative_tick = current_sc_time - current_tick + sysc_offset;
64 return final_relative_tick;
65 }
66 }

219 void runScheduler_()
220 {
221 // Start simulation -- align the schedulers
222 sparta_assert(sparta_scheduler_->nextEventTick() > 0);
223
224 // Align the schedulers. Sparta starts at tick 1
225 wait(sc_core::sc_time(double(1), sparta_sc_time_));
226
227 do {
228 // If the Sparta Scheduler has nothing to do, put it to sleep
229 if(sparta_scheduler_->nextEventTick() == Scheduler::INDEFINITE) {
230 sparta_scheduler_->registerForNotification<Scheduler::Tick,
231 SysCSpartaSchedulerAdapter,
232 &SysCSpartaSchedulerAdapter::wakeupAdapter_>(this, "item_scheduled");
233 wait(sc_wake_sparta_);
234 }
235
236 //
237 // Wait for SysC to get to the next event time on the
238 // Sparta Scheduler, then advance Sparta
239 //
240 const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
241 if(sparta_scheduler_->nextEventTick() >= sysc_time.value()) {
242 // Wait until the sysc scheduled catches up with Sparta
243 wait(sc_core::sc_time(double(sparta_scheduler_->nextEventTick() - sysc_time.value()),
244 sparta_sc_time_));
245 }
246
247 // Align to the posedge events in systemc
248 wait(sc_core::SC_ZERO_TIME);
249
250 // If given, schedule the user's SystemC query event to
251 // allow the Sparta user to check to see if the SystemC side
252 // of simulation is complete. This query can be as simple
253 // as asking the SystemC kernel if there are any events
254 // (sc_core::sc_pending_activity()) or asking the SystemC
255 // simulator if it's done.
256 if(sysc_query_event_ && sparta_scheduler_->getCurrentTick() >= next_sysc_event_fire_tick_)
257 {
258 next_sysc_event_fire_tick_ = sparta_scheduler_->getCurrentTick() + sysc_query_event_interval_;
259 sysc_query_event_->scheduleRelativeTick(1, sparta_scheduler_);
260 }
261
262 advanceSpartaScheduler_();
263
264 } while(!sparta_scheduler_->isFinished() || !sysc_simulation_done_);
265
266 // Stop simulation
267 sc_core::sc_stop();
268 sc_stop_called_ = true;
269 }
270
271 void advanceSpartaScheduler_()
272 {
273 const sc_core::sc_time sysc_time = sc_core::sc_time_stamp();
274
275 // The SystemC scheduler will always be exactly at the same
276 // tick as Sparta when this function is called. Following
277 // this rule allows safe assumptions in scheduling
278 // synchronization.
279 sparta_assert(sysc_time.value() == sparta_scheduler_->nextEventTick());
280 sparta_assert(sparta_scheduler_->nextEventTick() >= sparta_scheduler_->getCurrentTick());
281
282 const bool exacting_run = true;
283 const bool measure_scheduler_time = false; // no need to do this
284 // Run to the next event scheduled and then run that cycle
285 sparta_scheduler_->run(sparta_scheduler_->nextEventTick() - sparta_scheduler_->getCurrentTick() + 1,
286 exacting_run, measure_scheduler_time);
287
288 // Sparta should now be finished with the cycle at `sysc_time`
289 // and be exactly one cycle ahead of it.
290 sparta_assert(sysc_time.value() + 1 == sparta_scheduler_->getCurrentTick());
291 }
