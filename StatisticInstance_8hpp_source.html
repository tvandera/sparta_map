121 StatisticInstance(const StatisticDef* sd,
122 const CounterBase* ctr,
123 const ParameterBase* par,
124 const TreeNode* n,
125 std::vector<const TreeNode*>* used) :
126 StatisticInstance()
127 {
128 const StatisticDef* stat_def;
129 if(!sd){
130 stat_def = dynamic_cast<const StatisticDef*>(n);
131
132 sdef_ = stat_def;
133 }else{
134 sdef_ = stat_def = sd;
135 }
136 const CounterBase* counter;
137 if(!ctr){
138 counter = dynamic_cast<const CounterBase*>(n);
139 ctr_ = counter;
140 }else{
141 ctr_ = counter = ctr;
142 }
143 const ParameterBase* param;
144 if(!par){
145 param = dynamic_cast<const ParameterBase*>(n);
146 par_ = param;
147 }else{
148 par_ = param = par;
149 }
150
151 // Find the non-null argument
152 const TreeNode* node = n;
153 if(!node){
154 node = sd;
155 if(!node){
156 node = ctr;
157 if(!node){
158 node = par;
159 sparta_assert(node,
160 "StatisticInstance was constructed with all null arguments. "
161 "This is not allowed");
162 }
163 }
164 }
165 sparta_assert(int(nullptr != sdef_) + int(nullptr != ctr_) + int(nullptr != par_) == 1,
166 "Can only instantiate a StatisticInstance with either a StatisticDef, "
167 "a Counter, or a Parameter of any numeric type. Got Node: \"" << node->getLocation()
168 << "\". This node is not a stat, counter, or parameter.");
169
170 // Get the Scheduler as context
171 if(node->getClock()) {
172 scheduler_ = n->getClock()->getScheduler();
173 }
174
175 if(sdef_){
176 node_ref_ = stat_def->getWeakPtr();
177
178 std::vector<const TreeNode*>* local_used_ptr;
179 std::vector<const TreeNode*> temp_used;
180 if(!used){
181 local_used_ptr = &temp_used;
182 }else{
183 local_used_ptr = used;
184 }
185 stat_expr_ = sdef_->realizeExpression(*local_used_ptr);
186 if(stat_expr_.hasContent() == false){
187 throw SpartaException("Cannot construct StatisticInstance based on node ")
188 << stat_def->getLocation() << " because its expression: "
189 << stat_def->getExpression() << " parsed to an empty expression";
190 }
191 const auto & sub_stats_info = sdef_->getSubStatistics();
192 for (auto & sub_stat_creation_info : sub_stats_info) {
193 addSubStatistic_(sub_stat_creation_info);
194 }
195 }else if(ctr_){
196 node_ref_ = counter->getWeakPtr();
197 }else if(par_){
198 node_ref_ = param->getWeakPtr();
199 }else{
200 // Should not have been able to call constructor without 1 or
201 // the 3 args being non-null
202 throw SpartaException("Cannot instantiate a StatisticInstance without a statistic "
203 "definition or counter pointer");
204 }
205
206 start();
207
208 sparta_assert(false == node_ref_.expired());
209 }

454 void start() {
455 sparta_assert(direct_lookup_si_value_ == nullptr,
456 "You cannot call StatisticInstance::start() for an SI "
457 "that was recreated from a SimDB record");
458
459 start_tick_ = getScheduler_()->getElapsedTicks();
460 end_tick_ = Scheduler::INDEFINITE;
461
462 if(user_calculated_si_value_){
463 initial_.resetValue(user_calculated_si_value_->getCurrentValue());
464 result_ = NAN;
465 return;
466 }
467
468 if(sdef_ != nullptr){
469 if(node_ref_.expired() == true){
470 throw SpartaException("Cannot start() a StatisticInstance referring to a "
471 "destructed StatisticDef");
472 }
473 stat_expr_.start();
474 initial_.resetValue(0);
475 }else if(ctr_){
476 if(node_ref_.expired() == true){
477 throw SpartaException("Cannot start() a StatisticInstance referring to a "
478 "destructed Counter");
479 }
480 initial_.resetValue(ctr_->get());
481 }else if(par_){
482 if(node_ref_.expired() == true){
483 throw SpartaException("Cannot start() a StatisticInstance referring to a "
484 "destructed Parameter");
485 }
486 initial_.resetValue(par_->getDoubleValue());
487 }else{
488 stat_expr_.start();
489 }
490
491 // Clear result value
492 result_ = NAN;
493 }

503 void end(){
504 sparta_assert(direct_lookup_si_value_ == nullptr,
505 "You cannot call StatisticInstance::end() for an SI "
506 "that was recreated from a SimDB record");
507
508 end_tick_ = getScheduler_()->getElapsedTicks();
509
510 if(sdef_ != nullptr){
511 if(node_ref_.expired() == true){
512 throw SpartaException("Cannot end() a StatisticInstance referring to a "
513 "destructed StatisticDef");
514 }
515 stat_expr_.end();
516 }else if(ctr_ != nullptr){
517 if(node_ref_.expired() == true){
518 throw SpartaException("Cannot end() a StatisticInstance referring to a "
519 "destructed Counter");
520 }
521 // Do nothing to counter
522 }else if(par_ != nullptr){
523 if(node_ref_.expired() == true){
524 throw SpartaException("Cannot end() a StatisticInstance referring to a "
525 "destructed Parameter");
526 }
527 // Do nothing to Parameter
528 }else{
529 stat_expr_.end();
530 }
531
532 // Recompute result value
533 result_ = computeValue_();
534 }

616 double getValue() const {
617 if (direct_lookup_si_value_ != nullptr) {
618 return computeValue_();
619 }
620
621 if(end_tick_ < start_tick_){
622 throw ReversedStatisticRange("Range is reversed. End < start");
623 }
624
625 if(start_tick_ > getScheduler_()->getElapsedTicks()){
626 throw FutureStatisticRange("Range starts in the future at ") << start_tick_;
627 }
628
629 double value;
630 if(end_tick_ == Scheduler::INDEFINITE){
631 // Compute Value
632 value = computeValue_();
633 }
634
635 else if(end_tick_ > getScheduler_()->getElapsedTicks()){
636 // Rang ends in the future - probable because of a checkpoint
637 throw FutureStatisticRange("Range ends in the future at ") << end_tick_;
638 }
639
640 else {
641 // End tick <= current tick. Use pre-computed value because this
642 // window ended in the past
643 value = result_;
644 }
645
646 //Update any snapshot loggers that are listening for these updates
647 for (auto & logger : snapshot_loggers_) {
648 logger.takeSnapshot(value);
649 }
650 return value;
651 }
