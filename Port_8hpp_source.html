567 void bind(Port * in) override
568 {
569 if(in->getDirection() != Port::Direction::IN) {
570 throw SpartaException("ERROR: Attempt to bind an outty: '" + in->getName() +
571 "' to an outty: '" + getName() + "'");
572 }
573
574 if(!sync_port_) {
575 sparta_assert(in->getClock()->getFrequencyMhz() == getClock()->getFrequencyMhz(),
576 "Trying to bind two ports that are on different clocks with different freq. "
577 "Recommend using SyncPorts");
578 }
579
580 sparta_assert(!isAlreadyBound(in),
581 "Port: '" << getLocation()
582 << "' is already bound to '" << in->getLocation() << "' ");
583
584 InPort * inp = nullptr;
585 if((inp = dynamic_cast<InPort *>(in)) == 0) {
586 throw SpartaException("ERROR: Could not cast '" +
587 in->getName() + "' to an InPort for some reason...");
588 }
589
590 // This outport precedes the newly binded in port if the
591 // delay of the inport is zero and the modeler intends to
592 // use the OutPort as zero delay.
593 //
594 // The precedence that's established is the following, but
595 // only if the producer, the inport's internal events, and
596 // the consumer event are all in the same phase.
597 //
598 // ,-----------------------------------------------.
599 // | V
600 // producer -> [ inport internal delivery ]* -> consumer
601 // [ inport handler delivery (optional) ]
602 //
603 //
604 // * The inport -> consumer precedence is established
605 // during the consumer registration from a call to
606 // InPort::registerConsumerEvent()
607 //
608 if((inp->getPortDelay() == 0) && presume_zero_delay_) {
609 for(auto & pd : port_producers_)
610 {
611 for(auto & cons : inp->port_consumers_) {
612 sparta_assert(pd != cons,
613 "Somehow, someway, '" << pd->getLabel() << "' is registered "
614 << "as a producer of Port: '" << getLocation()
615 << "' and, at the same time, a consumer of Port: '"
616 << inp->getLocation() << "'");
617 if(pd->getSchedulingPhase() == cons->getSchedulingPhase()) {
618 std::string reason = "Port::bind(" + getLocation() + "->" + inp->getLocation() + ")";
619 pd->precedes(cons, reason);
620 }
621 }
622 inp->setProducerPrecedence_(pd);
623 }
624 }
625 bound_ports_.push_back(inp);
626 inp->setContinuing(continuing_);
627 inp->bind_(this);
628 }
