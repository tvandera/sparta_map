311 static void releaseRefCount_(RefCount *& ref_count)
312 {
313 if(0 == ref_count->count) {
314 BaseAllocator::MemBlockBase * memory_block =
315 static_cast<BaseAllocator::MemBlockBase *>(ref_count->mem_block);
316
317 if(SPARTA_EXPECT_TRUE(nullptr != memory_block)) {
318 memory_block->alloc->releaseObject_(memory_block);
319 }
320 else {
321 delete ref_count->p;
322 ref_count->p = nullptr;
323 }
324 // Make it go negative to show there are no
325 // SpartaSharedPointer objects using this ref_count.
326 // We can't set it to nullptr because it might be
327 // allocated on the heap and we need to ensure all
328 // SpartaWeakPointer objects are done with it too.
329 --(ref_count->count);
330 }
331
332 if((0 >= ref_count->count) && (0 == ref_count->wp_count))
333 {
334 BaseAllocator::MemBlockBase * memory_block =
335 static_cast<BaseAllocator::MemBlockBase *>(ref_count->mem_block);
336
337 if(SPARTA_EXPECT_TRUE(nullptr != memory_block)) {
338 memory_block->alloc->releaseBlock_(memory_block);
339 }
340 else {
341 delete ref_count;
342 }
343 ref_count = nullptr;
344 }
