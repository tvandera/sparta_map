201 PipelineCollector(const std::string& filepath, Scheduler::Tick heartbeat_interval,
202 const sparta::Clock* root_clk, const sparta::TreeNode* root,
203 Scheduler* scheduler=nullptr) :
204 Collector("PipelineCollector"),
205 scheduler_(scheduler != nullptr ? scheduler : root_clk->getScheduler()),
206 collector_events_(nullptr),
207 ev_heartbeat_(&collector_events_, Collector::getName() + "_heartbeat_event",
208 CREATE_SPARTA_HANDLER(PipelineCollector, performHeartBeat_), 0)
209 {
210 // Sanity check - pipeline collection cannot occur without a scheduler
211 sparta_assert(scheduler_);
212
213 ev_heartbeat_.setScheduleableClock(root_clk);
214 ev_heartbeat_.setScheduler(scheduler_);
215
216 // We need to reverse the dependency order for the PostTick GOP and ev_heartbeat_ so that every other
217 // event happens *before* ev_heartbeat_.
218 // Doing this ensures that we don't accidentally mark a 1-cycle transaction as continued.
219 DAG* dag = scheduler_->getDAG(); // Get a handle to the DAG
220 sparta_assert(dag);
221 Vertex* post_tick_gop = dag->getGOPoint("PostTick"); // Get a handle to the PostTick GOP
222 sparta_assert(post_tick_gop);
223
224 dag->unlink(ev_heartbeat_.getVertex(), post_tick_gop); // Undo the ev_heartbeat_ >> heartbeat_gop link
225 post_tick_gop->precedes(ev_heartbeat_); // Set post_tick_gop >> heartbeat_gop
226
227 sparta_assert(root != nullptr, "Pipeline Collection will not be able to create location file because it was passed a nullptr root treenode.");
228 sparta_assert(root->isFinalized(), "Pipeline collection cannot be constructed until the sparta tree has been finalized.");
229 // Assert that we got valid pointers necessary for pipeline collection.
230 sparta_assert(root_clk != nullptr, "Cannot construct PipelineCollector because root clock is a nullptr");
231 sparta_assert(scheduler_->isFinalized() == false, "Pipeline Collection cannot be instantiated after scheduler finalization -- it creates events");
232
233 // Initialize the clock/collectable map and find the fastest clock
234 const sparta::Clock* fastest_clk = nullptr;
235 std::function<void (const sparta::Clock*)> addClks;
236 addClks = [&addClks, this, &fastest_clk] (const sparta::Clock* clk)
237 {
238 if(clk != nullptr){
239 auto & u_p = clock_ctn_map_[clk];
240 for(uint32_t i = 0; i < NUM_SCHEDULING_PHASES; ++i) {
241 u_p[i].reset(new CollectablesByClock(clk, static_cast<SchedulingPhase>(i)));
242 }
243 for(const sparta::TreeNode* child : sparta::TreeNodePrivateAttorney::getAllChildren(clk)) {
244 const sparta::Clock* child_clk = dynamic_cast<const sparta::Clock*>(child);
245 if(child_clk){
246 auto clk_period = child_clk->getPeriod();
247 // If this clock has a non-1 period (i.e. not the root clock)
248 // AND there is either
249 // (A) no fastest clock yet
250 // or
251 // (B) this clock is a higher frequency than the fastest clock.
252 // then choose this as the fastest
253 if(clk_period != 1 && (!fastest_clk || (clk_period < fastest_clk->getPeriod()))) {
254 fastest_clk = child_clk;
255 }
256 addClks(child_clk);
257 }
258 }
259 }
260 };
261 addClks(root_clk);
262 if(fastest_clk == nullptr){
263 fastest_clk = root_clk;
264 }
265
266 // A multiple to multiply against the fastest clock when no heartbeat was set.
269 static const uint32_t heartbeat_multiplier = 200;
270 // round heartbeat using a multiple of the fastest clock if one was not set.
271 if (heartbeat_interval == 0)
272 {
273 sparta_assert(fastest_clk->getPeriod() != 0);
274 heartbeat_interval = fastest_clk->getPeriod() * heartbeat_multiplier; //round up to the nearest multiple of 100
275 heartbeat_interval = heartbeat_interval + (100 - (heartbeat_interval % 100));
276 // pipeViewer requires that intervals be a multiple of 100.
277 sparta_assert(heartbeat_interval % 100 == 0)
278 }
279
280 // We are gonna subtract one from the heartbeat_interval
281 // later.. Better be greater than one.
282 sparta_assert(heartbeat_interval > 1);
283 // Initialize some values.
284 filepath_ = filepath;
285 heartbeat_interval_ = heartbeat_interval;
286 closing_time_ = heartbeat_interval;
287 root_clk_ = root_clk;
288
289 ev_heartbeat_.setContinuing(false); // This event does not keep simulation going
290 }

334 void startCollection(sparta::TreeNode* starting_node)
335 {
336 if(collection_active_ == false)
337 {
338 // Create the outputter used for writing transactions to disk.
339 writer_.reset(new pipeViewer::Outputter(filepath_, heartbeat_interval_));
340
341 // We need to write an index on the start BEFORE any transactions have been written.
342 writer_->writeIndex();
343
344 // Write the clock information out
345 writeClockFile_();
346
347 // Open the locations file
348 location_writer_.reset(new pipeViewer::LocationFileWriter(filepath_));
349
350 // The reader needs heartbeat indexes up to the current
351 // collection point. This can happen on delayed pipeline
352 // collection. Start the heartbeats at the first interval
353 // as the Outputter class handles hb 0
354 last_heartbeat_ = 0;
355 const uint64_t num_hb = scheduler_->getCurrentTick()/heartbeat_interval_;
356 uint64_t cnt = 0;
357 while(cnt != num_hb) {
358 // write an index
359 writer_->writeIndex();
360 last_heartbeat_ += heartbeat_interval_;
361 ++cnt;
362 }
363
364 // Schedule a heartbeat at the next interval, offset from
365 // the current tick. For example, if the scheduler is at
366 // 6,600,456, then we want to schedule at 7,000,000 if the
367 // interval is 1M and the num_hb is 6
368 ev_heartbeat_.scheduleRelativeTick(((num_hb + 1) * heartbeat_interval_) -
369 scheduler_->getCurrentTick(), scheduler_);
370
371 collection_active_ = true;
372 }
373
374 *(location_writer_.get()) << (*starting_node);

376 // Recursively collect the start node and children
377 std::function<void (sparta::TreeNode* starting_node)> recursiveCollect;
378 recursiveCollect = [&recursiveCollect, this] (sparta::TreeNode* starting_node)
379 {
380 // First turn on this node if it's actually a CollectableTreeNode
381 CollectableTreeNode* c_node = dynamic_cast<CollectableTreeNode*>(starting_node);
382 if(c_node != nullptr) {
383 c_node->startCollecting(this);
384 registered_collectables_.insert(c_node);
385 }
386
387 // Recursive step. Go through the children and turn them on as well.
388 for(sparta::TreeNode* node : sparta::TreeNodePrivateAttorney::getAllChildren(starting_node))
389 {
390 recursiveCollect(node);
391 }
392 };
393
394 recursiveCollect(starting_node);
395 }

501 template<class R_Type>
502 void writeRecord(const R_Type& dat)
503 {
504 sparta_assert(collection_active_, "The pipeline head must be running in order to write a transaction");
505
506 // Make sure it's within the heartbeat window
507 sparta_assert(dat.time_End <= (last_heartbeat_ + heartbeat_interval_));
508 sparta_assert(dat.time_Start >= last_heartbeat_);
509
510
511 // Make sure transactions are exclusive.
512 // transaction [4999-5000] should be written BEFORE the index is written for transactions
513 // starting at 5000
514 sparta_assert(closing_time_ < heartbeat_interval_ // Ignore first heartbeat
515 || dat.time_Start >= closing_time_ - heartbeat_interval_,
516 "Attempted to write a pipeout record with exclusive start =("
517 << dat.time_Start << "), less than closing of previous interval"
518 << closing_time_ - heartbeat_interval_ );
519

521 // << dat.time_Start << " end: " << dat.time_End

588 void performHeartBeat_()
589 {
590 if(collection_active_) {
591 // Close all transactions
592 for(auto & ctn : registered_collectables_) {
593 if(ctn->isCollected()) {
594 ctn->restartRecord();
595 }
596 }
597
598 // write an index
599 writer_->writeIndex();
600
601 // Remember the last time we recorded a heartbeat
602 last_heartbeat_ = scheduler_->getCurrentTick();
603
604 // Schedule another heartbeat
605 ev_heartbeat_.schedule(heartbeat_interval_);
606 }
