19template <template<class, class...> class C, typename T>
20class VectorCopier
21{
22public:
23 typedef C<T> container_t;
24
25 const std::vector<container_t>& input_;
26
27 VectorCopier(const std::vector<container_t>& input)
28 : input_(input)
29 {;}
30
31 void deepCopy(std::vector<container_t>& output) const
32 {
33 for (const auto& el : input_){
34 output.push_back(el); // No copy of T because this unspecialized class does not know how
35 }
36 }
37
38 void extractRawCopy(std::vector<T*>& output) const
39 {
40 for (const container_t& el : input_){
41 output.push_back(el.get()); // Extract raw pointer
42 }
43 }
44};
