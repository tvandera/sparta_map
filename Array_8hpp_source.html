300 "Cannot operate on an uninitialized iterator.");

302 if(!sparta::notNull(array_)->getNextOldestIndex(index_)) {
303 index_ = (sparta::notNull(array_)->getOldestIndex());
304 }
305 }
306 else {

314 // Aged iterators will act circular if the array is
315 // filled with valid entries.
316 if(is_aged_)

318 // We've wrapped around -- we're done.
319 if(*this == sparta::notNull(array_)->abegin()) {

322 else {

324 // entry, try again.

328 }
329 }

332 // We've wrapped around and have hit begin. We're at
333 // the end now.
334 *this = sparta::notNull(array_)->end();
335 }
336 }

561 const_iterator getOldestIndex(const uint32_t nth=0) const
562 {
563 sparta_assert(ArrayT == ArrayType::AGED,
564 "Only AgedArray types have public member function getOldestIndex");
565 sparta_assert(nth < num_valid_,
566 "The array does not have enough elements to find the nth oldest index");
567
568 constexpr bool is_aged = true;
569 constexpr bool is_circular = false;
570 constexpr bool is_aged_walk = true;
571
572 // Since our data_list_ always adds new items to the
573 // front. The oldest data is actually kept at the end of
574 // the list. We can iterate from the end to find the nth
575 // oldest item.
576 auto it = aged_list_.rbegin();
577 uint32_t idx = invalid_entry_;
578 for(uint32_t i = 0; i <= nth; ++i)
579 {
580 idx = *it;
581 ++it;
582 }
583 // Double check that we are returning the user a valid
584 // index. We have failed if it isn't.
585 sparta_assert(isValid(idx));
586 return const_iterator(this, idx, is_aged, is_circular, is_aged_walk);
587 }

871 template<typename U>
872 void writeImpl_(const uint32_t idx, U&& dat)
873 {
874 sparta_assert(idx < num_entries_,
875 "Cannot write to an index outside the bounds of the array.");
876
877 // Since we are not timed. Write the data and validate it,
878 // then do pipeline collection.
879 new (array_.get() + idx) ArrayPosition(std::forward<U>(dat));
880 valid_index_set_.insert(idx);
881
882 // Timestamp the entry in the array, for fast age comparison between two indexes.
883 array_[idx].age_id = next_age_id_;
884 ++next_age_id_;
885
886 array_[idx].valid = true;
887 ++num_valid_;
888
889 // Maintain our age order if we are an aged array.
890 if(ArrayT == ArrayType::AGED)
891 {
892 // To maintain aged items, add the index to the front
893 // of a list.
894 aged_list_.push_front(idx);
895 array_[idx].list_pointer = aged_list_.begin();
896 }
897
898 // Update occupancy counter.
899 if(utilization_)
900 {
901 utilization_->setValue(num_valid_);
902 }
903 }
