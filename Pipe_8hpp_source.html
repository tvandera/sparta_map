539 void internalUpdate_() {
540 // Remove the head object
541 PipeEntry & pe_head = pipe_[getPhysicalStage_(num_entries_ - 1)];
542 if(pe_head.data.isValid()) {
543 --num_valid_;
544 }
545 pe_head.data.clearValid();
546
547 // Shift the pipe
548 tail_ = getPhysicalStage_(-1);
549
550 // Add the tail object
551 const PipeEntry & pe_tail = pipe_[tail_];
552 if(pe_tail.data.isValid()) {
553 ++num_valid_;
554 }
555 if((num_valid_ > 0) && perform_own_updates_) {
556 ev_update_.schedule();
557 }
558 }
