504 const Clock* getClock() {
505 std::vector<const Clock*> clocks;
506 getClocks(clocks);
507 if(clocks.size() == 0){
508 return nullptr;
509 }
510 const Clock* result = *(clocks.begin());
511 for(auto itr = clocks.begin()+1; itr != clocks.end(); ++itr){
512 if(*itr != result){
513 throw SpartaException("Multiple TreeNodes found with different clocks when "
514 "attempting to determine the clock associated with the "
515 "expression: ") << stringize();
516 }
517 }
518 return result;
519 }
